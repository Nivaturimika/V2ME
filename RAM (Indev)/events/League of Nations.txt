        name = "Cleanup Vote Check (Dumb Version)"

		any_country = {
			limit = {

		}
		any_country = {
			limit = {
				has_country_flag = LON_eject

			}
			country_event = 71033
		}
		any_country = {
			limit = {
				has_country_flag = LON_eject
				NOT = { UTI = { check_variable = { which = LON_Vote value = 0 } } }
			}
			country_event = 71034
		}
		any_country = {
			clr_country_flag = LON_Vote_For
			clr_country_flag = LON_Vote_Against
		}	
    }
}
