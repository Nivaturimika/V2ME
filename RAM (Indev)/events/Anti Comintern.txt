country_event = { # 
    id = 71104
    title = "ANTI COM Vote Check"
    desc = "Ai Checks if the variable is Positive. The Dyanamic Loc version would check the ratio."
    picture = "treaty"
    is_triggered_only = yes

    option = {
        name = "Cleanup Vote Check (Dumb Version)"
		UTI = { set_variable = { which = ANTI_COM_Vote value = -1 } }
		any_country = {
			limit = {
				has_country_flag = ANTI_COM_Vote_For
			}
			UTI = { change_variable = { which = ANTI_COM_Vote value = 1 } }
		}
		any_country = {
			limit = {
				has_country_flag = ANTI_COM_Vote_Against
			}
			UTI = { change_variable = { which = ANTI_COM_Vote value = -1 } }
		}
		any_country = {
			limit = {
				has_country_flag = ANTI_COM_Vote_eject
				UTI = { check_variable = { which = ANTI_COM_Vote value = 0 } }
			}
			country_event = 71105
		}
		any_country = {
			limit = {
				has_country_flag = ANTI_COM_Vote_eject
				NOT = { UTI = { check_variable = { which = ANTI_COM_Vote value = 0 } } }
			}
			country_event = 71106
		}
		any_country = {
			clr_country_flag = ANTI_COM_Vote_For
			clr_country_flag = ANTI_COM_Vote_Against
		}	
    }
}
