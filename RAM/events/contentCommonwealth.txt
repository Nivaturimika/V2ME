province_event = { # Hamilton British Empire Games
    id = 71007
    title = "EVTNAME71007"
    desc = "EVTDESC71007"
    picture = "commonwealth"

    trigger = {
		any_owned_province = { province_id = 55 } # Swap for hamilton
		year = 1930
		OR = {
			has_country_flag = british_commonwealth_of_nations_flag
			has_country_flag = british_commonwealth_of_nations_head_flag
		}
    }
	
	option = {
        name = "EVTOPTA71007"
		money = -97973
		prestige = 20
		55 = { add_province_modifier = { name = british_empire_games duration = 28 } }
		any_country = {
			limit = {
				OR = {
					has_country_flag = british_commonwealth_of_nations_flag
					has_country_flag = british_commonwealth_of_nations_head_flag
				}
			}
			country_event = { id = 71008 days = 1 }
		}
		
    }

    option = {
        name = "EVTOPTB71007"
		prestige = -20
		set_country_flag = rejected_empire_games
		random_country = {
			limit = {
				OR = {
					tag = ENG
					tag = SAF
					tag = AUS
					tag = NZL
					tag = NEW
				}
				exists = yes
				OR = {
					has_country_flag = british_commonwealth_of_nations_flag
					has_country_flag = british_commonwealth_of_nations_head_flag
				}
			}
			capital_scope = { province_event = { id = 71009 days = 1 } }
		}
    }
}
